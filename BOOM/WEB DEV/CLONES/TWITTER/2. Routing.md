more routes can be  added using the following structure:
- if 
- else if
- else if

```ts
var http = require('http'); // we need the http module

var server = http.createServer((req,res) => {
	console.log("joined");
	res.setHeader('Content-Type','text/html');
	const url = req.url;
	if(url == "/profile"){
		res.write('<p>This is the profile page</p>');
		return res.end();
	}
	else if(url == "/about"){
		res.write('<p>I m batman</p>');
		return res.end();
	}
	
	res.setHeader('Content-Type','text/html');
	res.write('<p>This is the profile page</p>');
	return res.end();


})
server.listen(3000);
```
